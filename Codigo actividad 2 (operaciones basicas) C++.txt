#include <iostream>
using namespace std;
int main()
{
    float num1, num2;
    
    cout << "Ingresa el primer numero: ";
    cin >> num1;
    
    cout << "Ingresa el segundo numero: ";
    cin >> num2;
    
    cout << "La suma es igual a: " << num1 + num2 << endl;
    cout << "La resta es igual a: " << num1 - num2 << endl;
    cout << "La multiplicacion es igual a: " << num1 * num2 << endl;
    
    if (num2 != 0) {
        cout << "La division es igual a: " << num1 / num2 << endl;
    } else {
        cout << "Lo siento no se puede dividir entre cero." << endl;
    }

    return 0;
}
